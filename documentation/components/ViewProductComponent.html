import { Component, inject } from '@angular/core';
import { ProductService } from './services/product.service';
import { Product } from './interfaces/product';
import { ProductCardFormat } from 'src/app/shared/ui/ui-product-card/types';

@Component({
  selector: 'app-view-product',
  templateUrl: './view-product.component.html',
  styleUrls: ['./view-product.component.scss']
})
export class ViewProductComponent {

  public productService = inject(ProductService);
  public selectedProduct:Product | undefined;

  public cardFormat:ProductCardFormat = 'standard';

  changeProductSelection( product:Product){
    this.selectedProduct = product;
  }

}


<ng-container [ngTemplateOutlet]="headerInfos"></ng-container>
<ng-container [ngTemplateOutlet]="controls"></ng-container>

<ng-template #headerInfos>
  <h1>Product Data</h1>
  <h2>Number of products {{ productService.products.length }}</h2>
</ng-template>

<ng-template #controls>
  <app-ui-button (click)=" productService.load() " *ngIf=" productService.products.length === 0 ">
    Click to load products
  </app-ui-button>

  <app-ui-button (click)=" cardFormat = (cardFormat === 'standard') ? 'compacted' : 'standard'  ">

</ng-template>

<ng-template #userFeedback>

      </app-ui-product-card>
